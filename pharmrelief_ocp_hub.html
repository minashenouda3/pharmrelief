// supabase/functions/ocp-sync/index.ts // Scheduled via: supabase functions deploy ocp-sync --schedule "0 23 * * *" import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' Deno.serve(async () => { const hour = new Date().getHours() // OCP only allows CSV export between 5pm and midnight if (hour < 17) return new Response('Outside OCP export window — retry after 5pm') // 1. Download fresh CSV from OCP public register search const res = await fetch('https://members.ocpinfo.com/tcpr/public/pr/en/', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ member_status: 'Active', export: 'csv' }) }) const csv = await res.text() const fresh = parseCSV(csv) // returns [{ocp, name, type, status, injection}] // 2. Load our current snapshot from Supabase const sb = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_KEY')!) const { data: existing } = await sb.from('ocp_snapshot').select('*') const existingMap = Object.fromEntries(existing.map(r => [r.ocp_number, r])) // 3. Diff — find new, changed, and removed records const changes = [] for (const r of fresh) { const prev = existingMap[r.ocp] if (!prev) { changes.push({ type: 'NEW_REGISTRATION', ...r }) } else if (prev.status !== r.status) { changes.push({ type: 'STATUS_CHANGE', ocp: r.ocp, name: r.name, from: prev.status, to: r.status }) } else if (prev.injection !== r.injection) { changes.push({ type: 'INJECTION_CHANGE', ocp: r.ocp, name: r.name, injection: r.injection }) } } // 4. Apply changes + trigger automations for (const change of changes) { await applyChange(sb, change) // see automation rules tab } // 5. Replace snapshot, log run, send alerts await sb.from('ocp_snapshot').delete().neq('id', 0) await sb.from('ocp_snapshot').insert(fresh.map(r => ({ ocp_number: r.ocp, ...r }))) await sb.from('sync_log').insert({ records: fresh.length, changes: changes.length }) if (changes.length > 0) await sendAlerts(changes) return new Response(JSON.stringify({ records: fresh.length, changes: changes.length })) })

// src/api/verify-ocp.ts // Called when pharmacist submits registration form export async function verifyOCPNumber(ocpNumber: string) { const response = await fetch( `https://members.ocpinfo.com/tcpr/public/pr/en/`, { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ registration_number: ocpNumber, member_status: 'All' }) } ) const html = await response.text() const result = { found: html.includes('Registration Number'), name: extractBetween(html, 'Name', ''), type: extractBetween(html, 'Registrant Type', ''), status: extractBetween(html, 'Member Status', ''), injection: html.includes('Injection Training: Yes'), conditions: html.includes('Terms, Conditions or Limitations'), discipline: html.includes('Discipline History'), } result.approved = ( result.found && result.status === 'Active' && !result.conditions && !result.discipline ) // Log to audit trail await logVerification(ocpNumber, result) // If suspended — notify agency manager if (result.found && result.status !== 'Active') { await alertAgencyManager(ocpNumber, result.status) } return result }

What OCP actually provides — complete honest guide

Exactly what is and isn't available from the Ontario College of Pharmacists for third-party integration

OCP eConnect is a newsletter, not a data API
OCP eConnect is an email newsletter published bi-weekly by the Ontario College of Pharmacists. It does not have an RSS feed, webhook, API endpoint, or programmatic subscription mechanism available to third parties. OCP does not offer any live data API for their member registry to external organisations. Everything in the OCP Data Hub is built on top of OCP's publicly accessible data only.
